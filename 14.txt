#include <iostream>
#include <cmath>
using namespace std;
 
int fib(int n){
    int f = 0;
    int res = 1;
    int f0 = 1;
    int i = 0;
    while (i <= n)
    {
        res = f;
        f = f0 + f;
        f0 = res;
        i++;
        if (n == f)
            return i;
    }
    if (n == 0)
    {
        return 0;
    }
    else 
    {
        cout << "Неверное значение!";
        return 0;
    }
}

    setlocale(LC_ALL, "Russian");

    cin >> n;
    cout << "K = " << fib(n);
    return 0;
}
