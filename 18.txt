#include <iostream>
#include <cmath>
using namespace std;


int main()
{
    setlocale(LC_ALL, "Rus");
    int n, i, imin, imax, dmin, dmax;
    cout << "Введите N: ";
    cin >> n;
    int a[n];
    dmin = 999999;
    dmax = 0;
    for (i = 1; i <= n; ++i)
    {
        cout << "a[" << i << "] = ";
        cin >> a[i];
    }
    for (i = 1; i <= n; ++i)
    {
        if (a[i] > dmax)
        {
            imax = i;
            dmax = a[i];
        }
        if (a[i] < dmin)
        {
            imin = i;
            dmin = a[i];
        }
    }
    for (i = 1; i <= n; ++i)
    {
        if (imin < imax)
        {
            for (i = imin + 1; i < imax; ++i)
            {
                a[i] = 0;
                cout << "a[" << i << "] = " << a[i] << endl;
            }
        }
        if (imin > imax)
        {
            for (i = imax + 1; i < imin; ++i)
            {
                a[i] = 0;
                cout << "a[" << i << "] = " << a[i] << endl;
            }
        }
    }
    for (i = 1; i <= n; ++i)
    {
        cout << "a[" << i << "] = " << a[i] << endl;
    }
}

#include <iostream>
#include <cmath>
using namespace std;


int main()
{
    setlocale(LC_ALL, "Rus");
    int n, i, d;
    cout << "Введите N: ";
    cin >> n;
    int a[n];
    for (i = 1; i <= n; ++i)
    {
        cout << "a[" << i << "] = ";
        cin >> a[i];
    }
    for (i = 3; i <= n; ++i)
    {
        if (a[i-1] >= a[i])
        {
            cout << "Значения от 2 до N должны идти по возрастанию!!!";
            return 0;
        }
    }
    for (i = 2; i <= n; ++i)
    {
        if (a[i-1] > a[i])
        {
            swap (a[i-1], a[i]);
        }
    }
    for (i = 1; i <= n; ++i)
    {
        cout << "a[" << i << "] = " << a[i] << endl;
    }
}
