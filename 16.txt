#include <iostream>
#include <cmath>
using namespace std;


int main()
{
    setlocale(LC_ALL, "Rus");
    int n, i, a, b, s;
    cout << "Введите N: ";
    cin >> n;
    cout << "Введите A: ";
    cin >> a;
    cout << "Введите B: ";
    cin >> b;
    if (n <= 2)
    {
        cout << "N должно быть больше 2!!!";
    }
    int arr[n];
    arr[0] = a;
    arr[1] = b;
    s = arr[0] + arr[1];
    cout << arr[0] << " ";
    cout << arr[1] << " ";
    for (i = 2; i <= n; ++i)
    {
        arr[i] = s;
        cout << arr[i] << " ";
        s += arr[i];
    }
}

#include <iostream>
#include <cmath>
using namespace std;


int main()
{
    setlocale(LC_ALL, "Rus");
    int n, i, j, k;
    cout << "Введите N: ";
    cin >> n;
    if (n <= 0)
    {
        cout << "N должно быть положительным!!!";
    }
    int a[n];
    for (i = 1; i <= n; ++i)
    {
        cout << "a[" << i << "] = ";
        cin >> a[i];
    }  
    j = k = 0;
    for (i = 1; i <= n; ++i)
    {
        if (i % 2 == 0)
        {
            cout << a[n-j] << " ";
            j += 1;
        }
        if (i % 2 == 1)
        {
            cout << a[i-k] << " ";
            k += 1;
        }
    }
    return 0;
}

#include <cmath>
using namespace std;


int main()
{
    setlocale(LC_ALL, "Rus");
    int n, i;
    cout << "Введите N: ";
    cin >> n;
    if (n <= 0)
    {
        cout << "N должно быть положительным!!!";
    }
    int a[n];
    for (i = 1; i <= n; ++i)
    {
        cout << "a[" << i << "] = ";
        cin >> a[i];
    }  
    for (i = 1; i <= n; i+=2)
    {
        cout << a[i] << " ";
    }
    if (n % 2 == 1)
    {
       n -= 1;  
    }
    for (i = n; i >= 2; i-=2)
    {
       cout << a[i] << " ";
    }
    return 0;
}
